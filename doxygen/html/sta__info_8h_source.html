224 /* Default value for maximum station inactivity. After AP_MAX_INACTIVITY has
225  * passed since last received frame from the station, a nullfunc data frame is
226  * sent to the station. If this frame is not acknowledged and no other frames
227  * have been received, the station will be disassociated after
228  * AP_DISASSOC_DELAY seconds. Similarly, the station will be deauthenticated
229  * after AP_DEAUTH_DELAY seconds has passed after disassociation. */
230 #define AP_MAX_INACTIVITY (5 * 60)
231 #define AP_DISASSOC_DELAY (3)
232 #define AP_DEAUTH_DELAY (1)
233 /* Number of seconds to keep STA entry with Authenticated flag after it has
234  * been disassociated. */
235 #define AP_MAX_INACTIVITY_AFTER_DISASSOC (1 * 30)
236 /* Number of seconds to keep STA entry after it has been deauthenticated. */
237 #define AP_MAX_INACTIVITY_AFTER_DEAUTH (1 * 5)
